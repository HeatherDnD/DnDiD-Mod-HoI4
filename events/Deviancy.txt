					create_wargoal = {
						target = YUG
						type = annex_everything
					}
				}
			}
		}		
	}
	
	option = { # Yugoslavia is our rightfull land!
		name = deviancy.2.c

		if = {
			limit = {
				NOT = { is_in_faction_with = DDD }

				hidden_effect = {
					country_event = { days = 1 id = deviancy.5 }
				}
			}	
			create_wargoal = {
				target = DDD
				type = annex_everything
			}
		}		
	}
